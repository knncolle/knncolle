25const Float_* l2norm(const Float_* ptr, size_t ndim, Float_* buffer) {
26 Float_ l2 = 0;
27 for (size_t d = 0; d < ndim; ++d) {
28 auto val = ptr[d];
29 l2 += val * val;
30 }
31
32 if (l2 == 0) {
33 return ptr;
34 }
