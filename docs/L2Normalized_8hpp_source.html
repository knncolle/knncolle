29template<typename Data_, typename Normalized_>
30void l2norm(const Data_* ptr, std::size_t ndim, Normalized_* buffer) {
31 Normalized_ l2 = 0;
32 for (std::size_t d = 0; d < ndim; ++d) {
33 Normalized_ val = ptr[d]; // cast to Normalized_ to avoid issues with integer overflow.
34 buffer[d] = val;
35 l2 += val * val;
36 }
37
38 if (l2 > 0) {
39 l2 = std::sqrt(l2);
40 for (std::size_t d = 0; d < ndim; ++d) {
41 buffer[d] /= l2;
42 }
43 }
44}
