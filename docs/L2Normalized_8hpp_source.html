58template<typename Data_, typename Normalized_>
59void l2norm(const Data_* ptr, std::size_t ndim, Normalized_* buffer) {
60 Normalized_ l2 = 0;
61 for (std::size_t d = 0; d < ndim; ++d) {
62 Normalized_ val = ptr[d]; // cast to Normalized_ to avoid issues with integer overflow.
63 buffer[d] = val;
64 l2 += val * val;
65 }
66
67 if (l2 > 0) {
68 l2 = std::sqrt(l2);
69 for (std::size_t d = 0; d < ndim; ++d) {
70 buffer[d] /= l2;
71 }
72 }
73}
