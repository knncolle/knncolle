37template<typename Data_, typename Normalized_>
38void l2norm(const Data_* ptr, std::size_t ndim, Normalized_* buffer) {
39 Normalized_ l2 = 0;
40 for (std::size_t d = 0; d < ndim; ++d) {
41 Normalized_ val = ptr[d]; // cast to Normalized_ to avoid issues with integer overflow.
42 buffer[d] = val;
43 l2 += val * val;
44 }
45
46 if (l2 > 0) {
47 l2 = std::sqrt(l2);
48 for (std::size_t d = 0; d < ndim; ++d) {
49 buffer[d] /= l2;
50 }
51 }
52}
