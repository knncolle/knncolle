25template<typename Float_>
26const Float_* l2norm(const Float_* ptr, size_t ndim, Float_* buffer) {
27 Float_ l2 = 0;
28 for (size_t d = 0; d < ndim; ++d) {
29 auto val = ptr[d];
30 l2 += val * val;
31 }
32
33 if (l2 == 0) {
34 return ptr;
35 }
36
37 l2 = std::sqrt(l2);
38 for (size_t d = 0; d < ndim; ++d) {
39 buffer[d] = ptr[d] / l2;
40 }
41 return buffer;
42}
