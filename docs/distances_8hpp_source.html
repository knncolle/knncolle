169struct CosineDistance {
183 template<typename Output_, typename DataX_, typename DataY_, typename Dim_>
184 static Output_ raw_distance(const DataX_* x, const DataY_* y, Dim_ num_dimensions) {
185 Output_ l2x = 0, l2y = 0;
186 for (Dim_ d = 0; d < num_dimensions; ++d) {
187 // casting to ensure consistent precision regardless of DataX_, DataY_.
188 Output_ curx = x[d], cury = y[d];
189 l2x += curx * curx;
190 l2y += cury * cury;
191 }
192
193 if (l2x == 0 || l2y == 0) {
194 // If only one of the L2 norms is all zero, the distance is 1. We
195 // don't bother to L2-normalize the all-zero vector and just
196 // compute the Euclidean distance as-is to the other
197 // (L2-normalized) vector, in which case the squared distance is
198 // just that vector's L2 divided by itself, a.k.a. 1.
199 //
200 // If both are all zero, the distance is zero.
201 return (l2x != l2y);
202 }
203
204 l2x = std::sqrt(l2x);
205 l2y = std::sqrt(l2y);
206
207 // While there are formulas that allow everything to be computed in a single
208 // pass, these are probably not numerically stable as you need to subtract
209 // the cosine similarity from 1; if it's close to 1, we'll have problems.
210 //
211 // To be honest, if you want to compute an efficient cosine distance,
212 // it is probably better to just L2-normalize the input data. But sometimes
213 // that's not possible so we provide this distance function instead.
214
215 Output_ output = 0;
216 for (Dim_ d = 0; d < num_dimensions; ++d, ++x, ++y) {
217 auto delta = static_cast<Output_>(*x) / l2x - static_cast<Output_>(*y) / l2y;
218 output += delta * delta;
219 }
220 return output;
221 }
