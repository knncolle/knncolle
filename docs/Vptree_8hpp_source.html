143 /* Adapted from http://stevehanov.ca/blog/index.php?id=130 */
144 static const Index_ LEAF = 0;

146 // Single node of a VP tree.
147 struct Node {
148 Float_ radius = 0;
149
150 // Original index of current vantage point, defining the center of the node.
151 Index_ index = 0;
152
153 // Node index of the next vantage point for all children closer than 'threshold' from the current vantage point.
154 // This must be > 0, as the first node in 'nodes' is the root and cannot be referenced from other nodes.
155 Index_ left = LEAF;
156
157 // Node index of the next vantage point for all children further than 'threshold' from the current vantage point.
158 // This must be > 0, as the first node in 'nodes' is the root and cannot be referenced from other nodes.
159 Index_ right = LEAF;
160 };
161
162 std::vector<Node> my_nodes;

164 typedef std::pair<Float_, Index_> DataPoint;
165
166 template<class Rng_>
167 Index_ build(Index_ lower, Index_ upper, const Store_* coords, std::vector<DataPoint>& items, Rng_& rng) {
168 /*
169 * We're assuming that lower < upper at each point within this
170 * recursion. This requires some protection at the call site
171 * when nobs = 0, see the constructor.
172 */
173
174 Index_ pos = my_nodes.size();
175 my_nodes.emplace_back();
176 Node& node = my_nodes.back(); // this is safe during recursion because we reserved 'nodes' already to the number of observations, see the constructor.
177
178 Index_ gap = upper - lower;
179 if (gap > 1) { // not yet at a leaft.
180
181 /* Choose an arbitrary point and move it to the start of the [lower, upper)
182 * interval in 'items'; this is our new vantage point.
183 *
184 * Yes, I know that the modulo method does not provide strictly
185 * uniform values but statistical correctness doesn't really matter
186 * here, and I don't want std::uniform_int_distribution's
187 * implementation-specific behavior.
188 */
189 Index_ i = (rng() % gap + lower);
190 std::swap(items[lower], items[i]);
191 const auto& vantage = items[lower];
192 node.index = vantage.second;
193 const Store_* vantage_ptr = coords + static_cast<size_t>(vantage.second) * my_long_ndim; // cast to avoid overflow.
194
195 // Compute distances to the new vantage point.
196 for (Index_ i = lower + 1; i < upper; ++i) {
197 const Store_* loc = coords + static_cast<size_t>(items[i].second) * my_long_ndim; // cast to avoid overflow.
198 items[i].first = Distance_::template raw_distance<Float_>(vantage_ptr, loc, my_dim);
199 }
200
201 // Partition around the median distance from the vantage point.
202 Index_ median = lower + gap/2;
203 Index_ lower_p1 = lower + 1; // excluding the vantage point itself, obviously.
204 std::nth_element(items.begin() + lower_p1, items.begin() + median, items.begin() + upper);
205
206 // Radius of the new node will be the distance to the median.
207 node.radius = Distance_::normalize(items[median].first);
208
209 // Recursively build tree.
210 if (lower_p1 < median) {
211 node.left = build(lower_p1, median, coords, items, rng);
212 }
213 if (median < upper) {
214 node.right = build(median, upper, coords, items, rng);
215 }
216
217 } else {
218 const auto& leaf = items[lower];
219 node.index = leaf.second;
220 }
221
222 return pos;
223 }

234 VptreePrebuilt(Dim_ num_dim, Index_ num_obs, std::vector<Store_> data) :
235 my_dim(num_dim),
236 my_obs(num_obs),
237 my_long_ndim(my_dim),
238 my_data(std::move(data))
239 {
240 if (num_obs) {
241 std::vector<DataPoint> items;
242 items.reserve(my_obs);
243 for (Index_ i = 0; i < my_obs; ++i) {
244 items.emplace_back(0, i);
245 }
246
247 my_nodes.reserve(my_obs);
248
249 // Statistical correctness doesn't matter (aside from tie breaking)
250 // so we'll just use a deterministically 'random' number to ensure
251 // we get the same ties for any given dataset but a different stream
252 // of numbers between datasets. Casting to get well-defined overflow.
253 uint64_t base = 1234567890, m1 = my_obs, m2 = my_dim;
254 std::mt19937_64 rand(base * m1 + m2);
255
256 build(0, my_obs, my_data.data(), items, rand);
257
258 // Resorting data in place to match order of occurrence within
259 // 'nodes', for better cache locality.
260 std::vector<uint8_t> used(my_obs);
261 std::vector<Store_> buffer(my_dim);
262 my_new_locations.resize(my_obs);
263 auto host = my_data.data();
264
265 for (Index_ o = 0; o < num_obs; ++o) {
266 if (used[o]) {
267 continue;
268 }
269
270 auto& current = my_nodes[o];
271 my_new_locations[current.index] = o;
272 if (current.index == o) {
273 continue;
274 }
275
276 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
277 std::copy_n(optr, my_dim, buffer.begin());
278 Index_ replacement = current.index;
279
280 do {
281 auto rptr = host + static_cast<size_t>(replacement) * my_long_ndim;
282 std::copy_n(rptr, my_dim, optr);
283 used[replacement] = 1;
284
285 const auto& next = my_nodes[replacement];
286 my_new_locations[next.index] = replacement;
287
288 optr = rptr;
289 replacement = next.index;
290 } while (replacement != o);
291
292 std::copy(buffer.begin(), buffer.end(), optr);
293 }
294 }

298 template<typename Query_>
299 void search_nn(Index_ curnode_index, const Query_* target, Float_& max_dist, internal::NeighborQueue<Index_, Float_>& nearest) const {
300 auto nptr = my_data.data() + static_cast<size_t>(curnode_index) * my_long_ndim; // cast to avoid overflow.
301 Float_ dist = Distance_::normalize(Distance_::template raw_distance<Float_>(nptr, target, my_dim));
302
303 // If current node is within the maximum distance:
304 const auto& curnode = my_nodes[curnode_index];
305 if (dist < max_dist) {
306 nearest.add(curnode.index, dist);
307 if (nearest.is_full()) {
308 max_dist = nearest.limit(); // update value of max_dist (farthest point in result list)
309 }
310 }
311
312 if (dist < curnode.radius) { // If the target lies within the radius of ball:
313 if (curnode.left != LEAF && dist - max_dist <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child first
314 search_nn(curnode.left, target, max_dist, nearest);
315 }
316
317 if (curnode.right != LEAF && dist + max_dist >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child
318 search_nn(curnode.right, target, max_dist, nearest);
319 }
320
321 } else { // If the target lies outsize the radius of the ball:
322 if (curnode.right != LEAF && dist + max_dist >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child first
323 search_nn(curnode.right, target, max_dist, nearest);
324 }
325
326 if (curnode.left != LEAF && dist - max_dist <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child
327 search_nn(curnode.left, target, max_dist, nearest);
328 }
329 }

332 template<bool count_only_, typename Query_, typename Output_>
333 void search_all(Index_ curnode_index, const Query_* target, Float_ threshold, Output_& all_neighbors) const {
334 auto nptr = my_data.data() + static_cast<size_t>(curnode_index) * my_long_ndim; // cast to avoid overflow.
335 Float_ dist = Distance_::normalize(Distance_::template raw_distance<Float_>(nptr, target, my_dim));
336
337 // If current node is within the maximum distance:
338 const auto& curnode = my_nodes[curnode_index];
339 if (dist <= threshold) {
340 if constexpr(count_only_) {
341 ++all_neighbors;
342 } else {
343 all_neighbors.emplace_back(dist, curnode.index);
344 }
345 }
346
347 if (dist < curnode.radius) { // If the target lies within the radius of ball:
348 if (curnode.left != LEAF && dist - threshold <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child first
349 search_all<count_only_>(curnode.left, target, threshold, all_neighbors);
350 }
351
352 if (curnode.right != LEAF && dist + threshold >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child
353 search_all<count_only_>(curnode.right, target, threshold, all_neighbors);
354 }
355
356 } else { // If the target lies outsize the radius of the ball:
357 if (curnode.right != LEAF && dist + threshold >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child first
358 search_all<count_only_>(curnode.right, target, threshold, all_neighbors);
359 }
360
361 if (curnode.left != LEAF && dist - threshold <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child
362 search_all<count_only_>(curnode.left, target, threshold, all_neighbors);
363 }
364 }
