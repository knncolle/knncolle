111 // Single node of a VP tree.
112 struct Node {
113 Float_ radius = 0;
114
115 // Original index of current vantage point, defining the center of the node.
116 Index_ index = 0;
117
118 // Node index of the next vantage point for all children closer than 'threshold' from the current vantage point.
119 // This must be > 0, as the first node in 'nodes' is the root and cannot be referenced from other nodes.
120 Index_ left = LEAF;
121
122 // Node index of the next vantage point for all children further than 'threshold' from the current vantage point.
123 // This must be > 0, as the first node in 'nodes' is the root and cannot be referenced from other nodes.
124 Index_ right = LEAF;
125 };

129 typedef std::pair<Float_, Index_> DataPoint;

131 template<class Rng_>
132 Index_ build(Index_ lower, Index_ upper, const Store_* coords, std::vector<DataPoint>& items, Rng_& rng) {
133 /*
134 * We're assuming that lower < upper at each point within this
135 * recursion. This requires some protection at the call site
136 * when nobs = 0, see the constructor.
137 */
138
139 Index_ pos = my_nodes.size();
140 my_nodes.emplace_back();
141 Node& node = my_nodes.back(); // this is safe during recursion because we reserved 'nodes' already to the number of observations, see the constructor.
142
143 Index_ gap = upper - lower;
144 if (gap > 1) { // not yet at a leaft.
145
146 /* Choose an arbitrary point and move it to the start of the [lower, upper)
147 * interval in 'items'; this is our new vantage point.
148 *
149 * Yes, I know that the modulo method does not provide strictly
150 * uniform values but statistical correctness doesn't really matter
151 * here, and I don't want std::uniform_int_distribution's
152 * implementation-specific behavior.
153 */
154 Index_ i = (rng() % gap + lower);
155 std::swap(items[lower], items[i]);
156 const auto& vantage = items[lower];
157 node.index = vantage.second;
158 const Store_* vantage_ptr = coords + static_cast<size_t>(vantage.second) * my_long_ndim; // cast to avoid overflow.
159
160 // Compute distances to the new vantage point.
161 for (Index_ i = lower + 1; i < upper; ++i) {
162 const Store_* loc = coords + static_cast<size_t>(items[i].second) * my_long_ndim; // cast to avoid overflow.
163 items[i].first = Distance_::template raw_distance<Float_>(vantage_ptr, loc, my_dim);
164 }
165
166 // Partition around the median distance from the vantage point.
167 Index_ median = lower + gap/2;
168 Index_ lower_p1 = lower + 1; // excluding the vantage point itself, obviously.
169 std::nth_element(items.begin() + lower_p1, items.begin() + median, items.begin() + upper);
170
171 // Radius of the new node will be the distance to the median.
172 node.radius = Distance_::normalize(items[median].first);
173
174 // Recursively build tree.
175 if (lower_p1 < median) {
176 node.left = build(lower_p1, median, coords, items, rng);
177 }
178 if (median < upper) {
179 node.right = build(median, upper, coords, items, rng);
180 }
181
182 } else {
183 const auto& leaf = items[lower];
184 node.index = leaf.second;
185 }
186
187 return pos;
188 }

199 VptreePrebuilt(Dim_ num_dim, Index_ num_obs, std::vector<Store_> data) :
200 my_dim(num_dim),

204 {
205 if (num_obs) {
206 std::vector<DataPoint> items;
207 items.reserve(my_obs);
208 for (Index_ i = 0; i < my_obs; ++i) {
209 items.emplace_back(0, i);
210 }
211
212 my_nodes.reserve(my_obs);
213
214 // Statistical correctness doesn't matter (aside from tie breaking)
215 // so we'll just use a deterministically 'random' number to ensure
216 // we get the same ties for any given dataset but a different stream
217 // of numbers between datasets. Casting to get well-defined overflow.
218 uint64_t base = 1234567890, m1 = my_obs, m2 = my_dim;
219 std::mt19937_64 rand(base * m1 + m2);
220
221 build(0, my_obs, my_data.data(), items, rand);
222
223 // Resorting data in place to match order of occurrence within
224 // 'nodes', for better cache locality.
225 std::vector<uint8_t> used(my_obs);
226 std::vector<Store_> buffer(my_dim);
227 my_new_locations.resize(my_obs);
228 auto host = my_data.data();
229
230 for (Index_ o = 0; o < num_obs; ++o) {
231 if (used[o]) {
232 continue;
233 }
234
235 auto& current = my_nodes[o];
236 my_new_locations[current.index] = o;
237 if (current.index == o) {
238 continue;
239 }
240
241 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
242 std::copy_n(optr, my_dim, buffer.begin());
243 Index_ replacement = current.index;
244
245 do {
246 auto rptr = host + static_cast<size_t>(replacement) * my_long_ndim;
247 std::copy_n(rptr, my_dim, optr);
248 used[replacement] = 1;
249
250 const auto& next = my_nodes[replacement];
251 my_new_locations[next.index] = replacement;
252
253 optr = rptr;
254 replacement = next.index;
255 } while (replacement != o);
256
257 std::copy(buffer.begin(), buffer.end(), optr);
258 }
259 }

263 template<typename Query_>
264 void search_nn(Index_ curnode_index, const Query_* target, Float_& max_dist, internal::NeighborQueue<Index_, Float_>& nearest) const {
265 auto nptr = my_data.data() + static_cast<size_t>(curnode_index) * my_long_ndim; // cast to avoid overflow.
266 Float_ dist = Distance_::normalize(Distance_::template raw_distance<Float_>(nptr, target, my_dim));
267
268 // If current node is within the maximum distance:
269 const auto& curnode = my_nodes[curnode_index];
270 if (dist < max_dist) {
271 nearest.add(curnode.index, dist);
272 if (nearest.is_full()) {
273 max_dist = nearest.limit(); // update value of max_dist (farthest point in result list)
274 }
275 }
276
277 if (dist < curnode.radius) { // If the target lies within the radius of ball:
278 if (curnode.left != LEAF && dist - max_dist <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child first
279 search_nn(curnode.left, target, max_dist, nearest);
280 }
281
282 if (curnode.right != LEAF && dist + max_dist >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child
283 search_nn(curnode.right, target, max_dist, nearest);
284 }
285
286 } else { // If the target lies outsize the radius of the ball:
287 if (curnode.right != LEAF && dist + max_dist >= curnode.radius) { // if there can still be neighbors outside the ball, recursively search right child first
288 search_nn(curnode.right, target, max_dist, nearest);
289 }
290
291 if (curnode.left != LEAF && dist - max_dist <= curnode.radius) { // if there can still be neighbors inside the ball, recursively search left child
292 search_nn(curnode.left, target, max_dist, nearest);
293 }
294 }
