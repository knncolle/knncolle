135 void search(const Data_* query, NeighborQueue<Index_, Distance_>& nearest) const {
136 auto copy = my_data.data();
137 Distance_ threshold_raw = std::numeric_limits<Distance_>::infinity();
138 for (Index_ x = 0; x < my_obs; ++x, copy += my_dim) {

140 if (dist_raw <= threshold_raw) {
