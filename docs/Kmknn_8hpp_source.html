166 KmknnPrebuilt(Dim_ num_dim, Index_ num_obs, std::vector<Store_> data, const KmknnOptions<Dim_, Index_, Store_>& options) :
167 my_dim(num_dim),
168 my_obs(num_obs),
169 my_long_ndim(my_dim),
170 my_data(std::move(data))
171 {
172 auto init = options.initialize_algorithm;
173 if (init == nullptr) {
174 init.reset(new kmeans::InitializeKmeanspp<kmeans::SimpleMatrix<Store_, Index_, Dim_>, Index_, Store_>);
175 }
176 auto refine = options.refine_algorithm;
177 if (refine == nullptr) {
178 refine.reset(new kmeans::RefineHartiganWong<kmeans::SimpleMatrix<Store_, Index_, Dim_>, Index_, Store_>);
179 }
180
181 Index_ ncenters = std::ceil(std::pow(my_obs, options.power));
182 my_centers.resize(static_cast<size_t>(ncenters) * my_long_ndim); // cast to avoid overflow problems.
183
184 kmeans::SimpleMatrix mat(my_dim, my_obs, my_data.data());
185 std::vector<Index_> clusters(my_obs);
186 auto output = kmeans::compute(mat, *init, *refine, ncenters, my_centers.data(), clusters.data());
187
188 // Removing empty clusters, e.g., due to duplicate points.
189 {
190 my_sizes.resize(ncenters);
191 std::vector<Index_> remap(ncenters);
192 Index_ survivors = 0;
193 for (Index_ c = 0; c < ncenters; ++c) {
194 if (output.sizes[c]) {
195 if (c > survivors) {
196 auto src = my_centers.begin() + static_cast<size_t>(c) * my_long_ndim; // cast to avoid overflow.
197 auto dest = my_centers.begin() + static_cast<size_t>(survivors) * my_long_ndim;
198 std::copy_n(src, my_dim, dest);
199 }
200 remap[c] = survivors;
201 my_sizes[survivors] = output.sizes[c];
202 ++survivors;
203 }
204 }
205
206 if (survivors < ncenters) {
207 for (auto& c : clusters) {
208 c = remap[c];
209 }
210 ncenters = survivors;
211 my_centers.resize(static_cast<size_t>(ncenters) * my_long_ndim);
212 }
213 }
214
215 my_offsets.resize(ncenters);
216 for (Index_ i = 1; i < ncenters; ++i) {
217 my_offsets[i] = my_offsets[i - 1] + my_sizes[i - 1];
218 }
219
220 // Organize points correctly; firstly, sorting by distance from the assigned center.
221 std::vector<std::pair<Float_, Index_> > by_distance(my_obs);
222 {
223 auto sofar = my_offsets;
224 auto host = my_data.data();
225 for (Index_ o = 0; o < my_obs; ++o) {
226 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
227 auto clustid = clusters[o];
228 auto cptr = my_centers.data() + static_cast<size_t>(clustid) * my_long_ndim;
229
230 auto& counter = sofar[clustid];
231 auto& current = by_distance[counter];
232 current.first = Distance_::normalize(Distance_::template raw_distance<Float_>(optr, cptr, my_dim));
233 current.second = o;
234
235 ++counter;
236 }
237
238 for (Index_ c = 0; c < ncenters; ++c) {
239 auto begin = by_distance.begin() + my_offsets[c];
240 std::sort(begin, begin + my_sizes[c]);
241 }
242 }
243
244 // Permuting in-place to mirror the reordered distances, so that the search is more cache-friendly.
245 {
246 auto host = my_data.data();
247 std::vector<uint8_t> used(my_obs);
248 std::vector<Store_> buffer(my_dim);
249 my_observation_id.resize(my_obs);
250 my_dist_to_centroid.resize(my_obs);
251 my_new_location.resize(my_obs);
252
253 for (Index_ o = 0; o < my_obs; ++o) {
254 if (used[o]) {
255 continue;
256 }
257
258 const auto& current = by_distance[o];
259 my_observation_id[o] = current.second;
260 my_dist_to_centroid[o] = current.first;
261 my_new_location[current.second] = o;
262 if (current.second == o) {
263 continue;
264 }
265
266 // We recursively perform a "thread" of replacements until we
267 // are able to find the home of the originally replaced 'o'.
268 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
269 std::copy_n(optr, my_dim, buffer.begin());
270 Index_ replacement = current.second;
271 do {
272 auto rptr = host + static_cast<size_t>(replacement) * my_long_ndim;
273 std::copy_n(rptr, my_dim, optr);
274 used[replacement] = 1;
275
276 const auto& next = by_distance[replacement];
277 my_observation_id[replacement] = next.second;
278 my_dist_to_centroid[replacement] = next.first;
279 my_new_location[next.second] = replacement;
280
281 optr = rptr;
282 replacement = next.second;
283 } while (replacement != o);
284
285 std::copy(buffer.begin(), buffer.end(), optr);
286 }
287 }
288
289 return;
290 }

293 template<typename Query_>
294 void search_nn(const Query_* target, internal::NeighborQueue<Index_, Float_>& nearest, std::vector<std::pair<Float_, Index_> >& center_order) const {
295 /* Computing distances to all centers and sorting them. The aim is to
296 * go through the nearest centers first, to get the shortest
297 * 'threshold' possible through the rest of the search.
298 */
299 center_order.clear();
300 size_t ncenters = my_sizes.size();
301 center_order.reserve(ncenters);
302 auto clust_ptr = my_centers.data();
303 for (size_t c = 0; c < ncenters; ++c, clust_ptr += my_dim) {
304 center_order.emplace_back(Distance_::template raw_distance<Float_>(target, clust_ptr, my_dim), c);
305 }
306 std::sort(center_order.begin(), center_order.end());
307
308 // Computing the distance to each center, and deciding whether to proceed for each cluster.
309 Float_ threshold_raw = -1;
310 for (const auto& curcent : center_order) {
311 const Index_ center = curcent.second;
312 const Float_ dist2center = Distance_::normalize(curcent.first);
313
314 const auto cur_nobs = my_sizes[center];
315 const Float_* dIt = my_dist_to_centroid.data() + my_offsets[center];
316 const Float_ maxdist = *(dIt + cur_nobs - 1);
317
318 Index_ firstcell = 0;
319#if KNNCOLLE_KMKNN_USE_UPPER
320 Float_ upper_bd = std::numeric_limits<Float_>::max();
321#endif
322
323 if (threshold_raw >= 0) {
324 const Float_ threshold = Distance_::normalize(threshold_raw);
325
326 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
327 * threshold + maxdist < dist2center
328 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
329 */
330 const Float_ lower_bd = dist2center - threshold;
331 if (maxdist < lower_bd) {
332 continue;
333 }
334
335 firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
336
337#if KNNCOLLE_KMKNN_USE_UPPER
338 /* This exploits the reverse triangle inequality, to ignore points where:
339 * threshold + dist2center < point-to-center distance
340 */
341 upper_bd = threshold + dist2center;
342#endif
343 }
344
345 const auto cur_start = my_offsets[center];
346 const auto* other_cell = my_data.data() + my_long_ndim * static_cast<size_t>(cur_start + firstcell); // cast to avoid overflow issues.
347 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_cell += my_dim) {
348#if KNNCOLLE_KMKNN_USE_UPPER
349 if (*(dIt + celldex) > upper_bd) {
350 break;
351 }
352#endif
353
354 auto dist2cell_raw = Distance_::template raw_distance<Float_>(target, other_cell, my_dim);
355 nearest.add(cur_start + celldex, dist2cell_raw);
356 if (nearest.is_full()) {
357 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
358#if KNNCOLLE_KMKNN_USE_UPPER
359 upper_bd = Distance_::normalize(threshold_raw) + dist2center;
360#endif
361 }
362 }
363 }
