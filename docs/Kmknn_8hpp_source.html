204 KmknnPrebuilt(Dim_ num_dim, Index_ num_obs, std::vector<Store_> data, const KmknnOptions<Dim_, Index_, Store_>& options) :
205 my_dim(num_dim),
206 my_obs(num_obs),
207 my_long_ndim(my_dim),
208 my_data(std::move(data))
209 {
210 auto init = options.initialize_algorithm;
211 if (init == nullptr) {
212 init.reset(new kmeans::InitializeKmeanspp<kmeans::SimpleMatrix<Store_, Index_, Dim_>, Index_, Store_>);
213 }
214 auto refine = options.refine_algorithm;
215 if (refine == nullptr) {
216 refine.reset(new kmeans::RefineHartiganWong<kmeans::SimpleMatrix<Store_, Index_, Dim_>, Index_, Store_>);
217 }
218
219 Index_ ncenters = std::ceil(std::pow(my_obs, options.power));
220 my_centers.resize(static_cast<size_t>(ncenters) * my_long_ndim); // cast to avoid overflow problems.
221
222 kmeans::SimpleMatrix mat(my_dim, my_obs, my_data.data());
223 std::vector<Index_> clusters(my_obs);
224 auto output = kmeans::compute(mat, *init, *refine, ncenters, my_centers.data(), clusters.data());
225
226 // Removing empty clusters, e.g., due to duplicate points.
227 {
228 my_sizes.resize(ncenters);
229 std::vector<Index_> remap(ncenters);
230 Index_ survivors = 0;
231 for (Index_ c = 0; c < ncenters; ++c) {
232 if (output.sizes[c]) {
233 if (c > survivors) {
234 auto src = my_centers.begin() + static_cast<size_t>(c) * my_long_ndim; // cast to avoid overflow.
235 auto dest = my_centers.begin() + static_cast<size_t>(survivors) * my_long_ndim;
236 std::copy_n(src, my_dim, dest);
237 }
238 remap[c] = survivors;
239 my_sizes[survivors] = output.sizes[c];
240 ++survivors;
241 }
242 }
243
244 if (survivors < ncenters) {
245 for (auto& c : clusters) {
246 c = remap[c];
247 }
248 ncenters = survivors;
249 my_centers.resize(static_cast<size_t>(ncenters) * my_long_ndim);
250 }
251 }
252
253 my_offsets.resize(ncenters);
254 for (Index_ i = 1; i < ncenters; ++i) {
255 my_offsets[i] = my_offsets[i - 1] + my_sizes[i - 1];
256 }
257
258 // Organize points correctly; firstly, sorting by distance from the assigned center.
259 std::vector<std::pair<Float_, Index_> > by_distance(my_obs);
260 {
261 auto sofar = my_offsets;
262 auto host = my_data.data();
263 for (Index_ o = 0; o < my_obs; ++o) {
264 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
265 auto clustid = clusters[o];
266 auto cptr = my_centers.data() + static_cast<size_t>(clustid) * my_long_ndim;
267
268 auto& counter = sofar[clustid];
269 auto& current = by_distance[counter];
270 current.first = Distance_::normalize(Distance_::template raw_distance<Float_>(optr, cptr, my_dim));
271 current.second = o;
272
273 ++counter;
274 }
275
276 for (Index_ c = 0; c < ncenters; ++c) {
277 auto begin = by_distance.begin() + my_offsets[c];
278 std::sort(begin, begin + my_sizes[c]);
279 }
280 }
281
282 // Permuting in-place to mirror the reordered distances, so that the search is more cache-friendly.
283 {
284 auto host = my_data.data();
285 std::vector<uint8_t> used(my_obs);
286 std::vector<Store_> buffer(my_dim);
287 my_observation_id.resize(my_obs);
288 my_dist_to_centroid.resize(my_obs);
289 my_new_location.resize(my_obs);
290
291 for (Index_ o = 0; o < my_obs; ++o) {
292 if (used[o]) {
293 continue;
294 }
295
296 const auto& current = by_distance[o];
297 my_observation_id[o] = current.second;
298 my_dist_to_centroid[o] = current.first;
299 my_new_location[current.second] = o;
300 if (current.second == o) {
301 continue;
302 }
303
304 // We recursively perform a "thread" of replacements until we
305 // are able to find the home of the originally replaced 'o'.
306 auto optr = host + static_cast<size_t>(o) * my_long_ndim;
307 std::copy_n(optr, my_dim, buffer.begin());
308 Index_ replacement = current.second;
309 do {
310 auto rptr = host + static_cast<size_t>(replacement) * my_long_ndim;
311 std::copy_n(rptr, my_dim, optr);
312 used[replacement] = 1;
313
314 const auto& next = by_distance[replacement];
315 my_observation_id[replacement] = next.second;
316 my_dist_to_centroid[replacement] = next.first;
317 my_new_location[next.second] = replacement;
318
319 optr = rptr;
320 replacement = next.second;
321 } while (replacement != o);
322
323 std::copy(buffer.begin(), buffer.end(), optr);
324 }
325 }
326
327 return;
328 }

331 template<typename Query_>
332 void search_nn(const Query_* target, internal::NeighborQueue<Index_, Float_>& nearest, std::vector<std::pair<Float_, Index_> >& center_order) const {
333 /* Computing distances to all centers and sorting them. The aim is to
334 * go through the nearest centers first, to try to get the shortest
335 * threshold (i.e., 'nearest.limit()') possible at the start;
336 * this allows us to skip searches of the later clusters.
337 */
338 center_order.clear();
339 size_t ncenters = my_sizes.size();
340 center_order.reserve(ncenters);
341 auto clust_ptr = my_centers.data();
342 for (size_t c = 0; c < ncenters; ++c, clust_ptr += my_dim) {
343 center_order.emplace_back(Distance_::template raw_distance<Float_>(target, clust_ptr, my_dim), c);
344 }
345 std::sort(center_order.begin(), center_order.end());
346
347 // Computing the distance to each center, and deciding whether to proceed for each cluster.
348 Float_ threshold_raw = -1;
349 for (const auto& curcent : center_order) {
350 const Index_ center = curcent.second;
351 const Float_ dist2center = Distance_::normalize(curcent.first);
352
353 const auto cur_nobs = my_sizes[center];
354 const Float_* dIt = my_dist_to_centroid.data() + my_offsets[center];
355 const Float_ maxdist = *(dIt + cur_nobs - 1);
356
357 Index_ firstcell = 0;
358#if KNNCOLLE_KMKNN_USE_UPPER
359 Float_ upper_bd = std::numeric_limits<Float_>::max();
360#endif
361
362 if (threshold_raw >= 0) {
363 const Float_ threshold = Distance_::normalize(threshold_raw);
364
365 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
366 * threshold + maxdist < dist2center
367 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
368 */
369 const Float_ lower_bd = dist2center - threshold;
370 if (maxdist < lower_bd) {
371 continue;
372 }
373
374 firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
375
376#if KNNCOLLE_KMKNN_USE_UPPER
377 /* This exploits the reverse triangle inequality, to ignore points where:
378 * threshold + dist2center < point-to-center distance
379 */
380 upper_bd = threshold + dist2center;
381#endif
382 }
383
384 const auto cur_start = my_offsets[center];
385 const auto* other_cell = my_data.data() + my_long_ndim * static_cast<size_t>(cur_start + firstcell); // cast to avoid overflow issues.
386 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_cell += my_dim) {
387#if KNNCOLLE_KMKNN_USE_UPPER
388 if (*(dIt + celldex) > upper_bd) {
389 break;
390 }
391#endif
392
393 auto dist2cell_raw = Distance_::template raw_distance<Float_>(target, other_cell, my_dim);
394 nearest.add(cur_start + celldex, dist2cell_raw);
395 if (nearest.is_full()) {
396 threshold_raw = nearest.limit(); // Shrinking the threshold, if an earlier NN has been found.
397#if KNNCOLLE_KMKNN_USE_UPPER
398 upper_bd = Distance_::normalize(threshold_raw) + dist2center;
399#endif
400 }
401 }
402 }

405 template<bool count_only_, typename Query_, typename Output_>
406 void search_all(const Query_* target, Float_ threshold, Output_& all_neighbors) const {
407 Float_ threshold_raw = threshold * threshold;
408
409 /* Computing distances to all centers. We don't sort them here
410 * because the threshold is constant so there's no point.
411 */
412 Index_ ncenters = my_sizes.size();
413 auto center_ptr = my_centers.data();
414 for (Index_ center = 0; center < ncenters; ++center, center_ptr += my_dim) {
415 const Float_ dist2center = Distance_::normalize(Distance_::template raw_distance<Float_>(target, center_ptr, my_dim));
416
417 auto cur_nobs = my_sizes[center];
418 const Float_* dIt = my_dist_to_centroid.data() + my_offsets[center];
419 const Float_ maxdist = *(dIt + cur_nobs - 1);
420
421 /* The conditional expression below exploits the triangle inequality; it is equivalent to asking whether:
422 * threshold + maxdist < dist2center
423 * All points (if any) within this cluster with distances above 'lower_bd' are potentially countable.
424 */
425 const Float_ lower_bd = dist2center - threshold;
426 if (maxdist < lower_bd) {
427 continue;
428 }
429
430 Index_ firstcell = std::lower_bound(dIt, dIt + cur_nobs, lower_bd) - dIt;
431#if KNNCOLLE_KMKNN_USE_UPPER
432 /* This exploits the reverse triangle inequality, to ignore points where:
433 * threshold + dist2center < point-to-center distance
434 */
435 Float_ upper_bd = threshold + dist2center;
436#endif
437
438 const auto cur_start = my_offsets[center];
439 auto other_ptr = my_data.data() + my_long_ndim * static_cast<size_t>(cur_start + firstcell); // cast to avoid overflow issues.
440 for (auto celldex = firstcell; celldex < cur_nobs; ++celldex, other_ptr += my_dim) {
441#if KNNCOLLE_KMKNN_USE_UPPER
442 if (*(dIt + celldex) > upper_bd) {
443 break;
444 }
445#endif
446
447 auto dist2cell_raw = Distance_::template raw_distance<Float_>(target, other_ptr, my_dim);
448 if (dist2cell_raw <= threshold_raw) {
449 if constexpr(count_only_) {
450 ++all_neighbors;
451 } else {
452 all_neighbors.emplace_back(dist2cell_raw, cur_start + celldex);
453 }
454 }
455 }
