119Prebuilt<Index_, Data_, Distance_>* load_prebuilt_raw(const std::string& prefix) {
120 const auto alg_path = prefix + "ALGORITHM";
121 const auto algorithm = quick_load_as_string(alg_path);
122
123 const auto& reg = load_prebuilt_registry<Index_, Data_, Distance_>();
124 auto it = reg.find(algorithm);
125 if (it == reg.end()) {

129 return (it->second)(prefix);
