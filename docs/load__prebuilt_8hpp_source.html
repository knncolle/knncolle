133template<typename Index_, typename Data_, typename Distance_>
134Prebuilt<Index_, Data_, Distance_>* load_prebuilt_raw(const std::string& prefix) {
135 const auto alg_path = prefix + "ALGORITHM";
136 const auto algorithm = quick_load_as_string(alg_path);

138 const auto& reg = load_prebuilt_registry<Index_, Data_, Distance_>();
139 auto it = reg.find(algorithm);
