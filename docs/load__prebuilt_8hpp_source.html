134template<typename Index_, typename Data_, typename Distance_>
135Prebuilt<Index_, Data_, Distance_>* load_prebuilt_raw(const std::filesystem::path& dir) {
136 const auto alg_path = dir / "ALGORITHM";
137 const auto algorithm = quick_load_as_string(alg_path);
138
139 const auto& reg = load_prebuilt_registry<Index_, Data_, Distance_>();
140 auto it = reg.find(algorithm);
141 if (it == reg.end()) {
142 throw LoadPrebuiltNotFoundError(algorithm, alg_path);
143 }
144
145 return (it->second)(dir);
146}
